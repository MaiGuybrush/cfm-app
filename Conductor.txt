export default function App() {
  const [currentSlide, setCurrentSlide] = useState(0);

  const nextSlide = useCallback(() => {
    setCurrentSlide((prev) => (prev < slides.length - 1 ? prev + 1 : prev));
  }, []);

  const prevSlide = useCallback(() => {
    setCurrentSlide((prev) => (prev > 0 ? prev - 1 : prev));
  }, []);

  useEffect(() => {
    const handleKeyDown = (e) => {
      if (e.key === 'ArrowRight' || e.key === ' ') nextSlide();
      if (e.key === 'ArrowLeft') prevSlide();
    };
    window.addEventListener('keydown', handleKeyDown);
    return () => window.removeEventListener('keydown', handleKeyDown);
  }, [nextSlide, prevSlide]);

  const renderSlide = () => {
    const slide = slides[currentSlide];

    switch (slide.type) {
      case 'title':
        return (
          <div className="flex flex-col items-center justify-center h-full text-center px-8">
            <div className="animate-bounce">{slide.icon}</div>
            <h1 className="text-5xl font-bold text-slate-100 mb-4 tracking-tight leading-tight">
              {slide.title}
            </h1>
            <div className="h-1 w-32 bg-blue-500 mb-6 rounded-full" />
            <h2 className="text-2xl text-slate-400 mb-12 font-light">
              {slide.subtitle}
            </h2>
            <div className="text-slate-500 space-y-1">
              <p className="font-medium text-slate-300">{slide.presenter}</p>
              <p className="text-sm">{slide.date}</p>
            </div>
          </div>
        );

      case 'content':
        return (
          <div className="p-12 h-full flex flex-col">
            <h2 className="text-3xl font-bold text-slate-100 mb-8 border-l-4 border-blue-500 pl-4">
              {slide.title}
            </h2>
            <div className="grid grid-cols-1 gap-6 flex-grow items-center max-w-4xl mx-auto">
              {slide.content.map((item, i) => (
                <div 
                  key={i} 
                  className={`${item.isHeader ? 'bg-slate-800/50 p-6 rounded-xl border border-slate-700' : 'flex items-start space-x-4 p-2'}`}
                >
                  {item.icon && <div className="mt-1">{item.icon}</div>}
                  <div>
                    <h3 className={`font-bold ${item.isHeader ? 'text-2xl text-blue-400 mb-2' : 'text-xl text-slate-200'}`}>
                      {item.label}
                    </h3>
                    <p className="text-slate-400 leading-relaxed">{item.desc}</p>
                  </div>
                </div>
              ))}
            </div>
          </div>
        );

      case 'framework':
        const isBlue = slide.color === 'blue';
        return (
          <div className="p-12 h-full flex flex-col">
            <div className="flex justify-between items-start mb-6">
              <h2 className="text-3xl font-bold text-slate-100 border-l-4 border-blue-500 pl-4">
                {slide.title}
              </h2>
              <span className={`px-4 py-1 rounded-full text-sm font-bold ${isBlue ? 'bg-blue-500/20 text-blue-400' : 'bg-emerald-500/20 text-emerald-400'}`}>
                {slide.philosophy}
              </span>
            </div>
            
            <p className="text-xl text-slate-300 mb-8 bg-slate-800/30 p-4 rounded-lg italic">
              「{slide.concept}」
            </p>

            <div className="grid grid-cols-3 gap-4 mb-8">
              {slide.files.map((file, i) => (
                <div key={i} className="bg-slate-800 p-4 rounded-xl border border-slate-700 hover:border-slate-500 transition-colors">
                  <div className="flex items-center space-x-2 mb-2">
                    <FileCode className={`w-5 h-5 ${isBlue ? 'text-blue-400' : 'text-emerald-400'}`} />
                    <span className="font-mono text-slate-200">{file.name}</span>
                  </div>
                  <p className="text-xs text-slate-500 mb-1">{file.role}</p>
                  <p className="text-sm text-slate-400">{file.detail}</p>
                </div>
              ))}
            </div>

            <div className="space-y-4">
              <h4 className="text-sm font-bold text-slate-500 uppercase tracking-widest">優勢：</h4>
              <div className="grid grid-cols-2 gap-4">
                {slide.advantages.map((adv, i) => (
                  <div key={i} className="flex items-center space-x-3 bg-slate-800/40 p-3 rounded-lg">
                    <CheckCircle2 className={`w-5 h-5 ${isBlue ? 'text-blue-400' : 'text-emerald-400'}`} />
                    <span className="text-slate-300">{adv}</span>
                  </div>
                ))}
              </div>
            </div>
          </div>
        );

      case 'comparison':
        return (
          <div className="p-12 h-full flex flex-col">
            <h2 className="text-3xl font-bold text-slate-100 mb-10 border-l-4 border-blue-500 pl-4">
              {slide.title}
            </h2>
            <div className="overflow-hidden rounded-xl border border-slate-700">
              <table className="w-full text-left">
                <thead className="bg-slate-800 text-slate-400 text-sm uppercase">
                  <tr>
                    <th className="px-6 py-4">特性指標</th>
                    <th className="px-6 py-4 text-blue-400">Gemini Conductor</th>
                    <th className="px-6 py-4 text-emerald-400">Spec-Kit</th>
                  </tr>
                </thead>
                <tbody className="divide-y divide-slate-700">
                  {slide.data.map((row, i) => (
                    <tr key={i} className="hover:bg-slate-800/30 transition-colors">
                      <td className="px-6 py-4 font-bold text-slate-400">{row.trait}</td>
                      <td className="px-6 py-4 text-slate-200">{row.conductor}</td>
                      <td className="px-6 py-4 text-slate-200">{row.speckt}</td>
                    </tr>
                  ))}
                </tbody>
              </table>
            </div>
          </div>
        );

      case 'metrics':
        return (
          <div className="p-12 h-full flex flex-col">
            <h2 className="text-3xl font-bold text-slate-100 mb-8 border-l-4 border-blue-500 pl-4">
              {slide.title}
            </h2>
            <div className="grid grid-cols-1 gap-6">
              {slide.metrics.map((m, i) => (
                <div key={i} className="bg-slate-800/40 p-6 rounded-2xl border border-slate-700 relative overflow-hidden group">
                  <div className="flex flex-col md:flex-row md:items-center space-y-4 md:space-y-0">
                    <div className="md:w-1/4">
                      <h3 className="text-xl font-bold text-blue-400">{m.label}</h3>
                    </div>
                    <div className="md:w-3/4 grid grid-cols-1 md:grid-cols-2 gap-4">
                      <div className="bg-blue-500/5 p-3 rounded-lg border border-blue-500/20">
                        <span className="text-xs text-blue-500 font-bold block mb-1">CONDUCTOR</span>
                        <p className="text-sm text-slate-300">{m.conductor}</p>
                      </div>
                      <div className="bg-emerald-500/5 p-3 rounded-lg border border-emerald-500/20">
                        <span className="text-xs text-emerald-500 font-bold block mb-1">SPEC-KIT</span>

                    </div>
                  </div>
                </div>
              ))}
            </div>
          </div>
        );

      case 'matrix':
        return (
          <div className="p-12 h-full flex flex-col">

            </h2>
            <div className="grid grid-cols-2 gap-8 flex-grow">
              {slide.options.map((opt, i) => (
                <div key={i} className={`p-8 rounded-3xl border-2 ${i === 0 ? 'border-blue-500/30 bg-blue-500/5' : 'border-emerald-500/30 bg-emerald-500/5'} flex flex-col`}>
                  <h3 className={`text-3xl font-black mb-6 ${i === 0 ? 'text-blue-400' : 'text-emerald-400'}`}>
                    {opt.name}
                  </h3>
                  <div className="space-y-4 mb-8 flex-grow">
                    <p className="text-slate-400 text-sm font-bold uppercase tracking-wider">最佳適用：</p>
                    <ul className="space-y-3">
                      {opt.fit.map((item, idx) => (
                        <li key={idx} className="flex items-center space-x-3 text-slate-200">
                          <CheckCircle2 className={`w-5 h-5 ${i === 0 ? 'text-blue-400' : 'text-emerald-400'}`} />
                          <span>{item}</span>
                        </li>
                      ))}
                    </ul>
                  </div>
                  <div className={`p-4 rounded-xl ${i === 0 ? 'bg-blue-500/10' : 'bg-emerald-500/10'} italic text-slate-300 border-l-4 ${i === 0 ? 'border-blue-400' : 'border-emerald-400'}`}>
                    解決：{opt.painPoint}
                  </div>
                </div>
              ))}
            </div>
          </div>
        );

      case 'summary':
        return (
          <div className="p-12 h-full flex flex-col">
            <h2 className="text-3xl font-bold text-slate-100 mb-8 border-l-4 border-blue-500 pl-4">
              {slide.title}
            </h2>
            <div className="space-y-6 mb-12">
              {slide.items.map((item, i) => (
                <div key={i} className="flex items-start space-x-4 bg-slate-800/30 p-4 rounded-xl border border-slate-700">
                  <div className="bg-blue-500/20 p-2 rounded-lg">
                    {i === 2 ? <ArrowRightLeft className="text-blue-400" /> : <Target className="text-blue-400" />}
                  </div>
                  <div>
                    <span className="font-bold text-slate-200 block text-lg">{item.label}</span>
                    <p className="text-slate-400">{item.text}</p>
                  </div>
                </div>
              ))}
            </div>
            <div className="mt-auto bg-gradient-to-r from-blue-600 to-indigo-600 p-8 rounded-2xl text-center shadow-2xl shadow-blue-500/20">
              <p className="text-2xl font-bold text-white mb-2 underline decoration-blue-300 underline-offset-4">核心建議</p>
              <p className="text-blue-100 text-xl">{slide.final}</p>
            </div>
          </div>
        );

      default:
        return null;
    }
  };

  return (
    <div className="min-h-screen bg-[#0f172a] text-slate-200 flex items-center justify-center font-sans overflow-hidden">
      {/* Background decoration */}
      <div className="fixed top-0 left-0 w-full h-full pointer-events-none opacity-20">
        <div className="absolute top-[-10%] left-[-10%] w-[40%] h-[40%] bg-blue-500 blur-[120px] rounded-full" />
        <div className="absolute bottom-[-10%] right-[-10%] w-[40%] h-[40%] bg-indigo-600 blur-[120px] rounded-full" />
      </div>

      <div className="relative w-full max-w-6xl aspect-video bg-slate-900 shadow-2xl rounded-3xl overflow-hidden border border-slate-800 flex flex-col">
        {/* Progress bar */}
        <div className="h-1.5 w-full bg-slate-800">
          <div 
            className="h-full bg-blue-500 transition-all duration-300"
            style={{ width: `${((currentSlide + 1) / slides.length) * 100}%` }}
          />
        </div>

        {/* Content area */}
        <div className="flex-grow overflow-y-auto">
          {renderSlide()}
        </div>

        {/* Navigation Bar */}
        <div className="h-16 px-6 border-t border-slate-800 flex items-center justify-between bg-slate-900/50 backdrop-blur-md">
          <div className="flex items-center space-x-4">
            <span className="text-xs font-mono text-slate-500 bg-slate-800 px-2 py-1 rounded">
              {currentSlide + 1} / {slides.length}
            </span>
            <span className="text-xs text-slate-600 hidden md:inline-block">
              使用鍵盤方向鍵或點擊導航
            </span>
          </div>

          <div className="flex items-center space-x-2">
            <button 
              onClick={prevSlide}
              disabled={currentSlide === 0}
              className="p-2 rounded-lg hover:bg-slate-800 disabled:opacity-30 disabled:cursor-not-allowed transition-colors"
            >
              <ChevronLeft className="w-6 h-6" />
            </button>
            <button 
              onClick={nextSlide}
              disabled={currentSlide === slides.length - 1}
              className="p-2 rounded-lg hover:bg-slate-800 disabled:opacity-30 disabled:cursor-not-allowed transition-colors"
            >
              <ChevronRight className="w-6 h-6" />
            </button>
          </div>
        </div>
      </div>
      
      {/* Mobile Controls Overlay */}
      <div className="md:hidden fixed bottom-4 right-4 flex space-x-2">
         <button onClick={prevSlide} className="bg-blue-600 p-4 rounded-full shadow-lg"><ChevronLeft /></button>
         <button onClick={nextSlide} className="bg-blue-600 p-4 rounded-full shadow-lg"><ChevronRight /></button>
      </div>
    </div>
  );
}
